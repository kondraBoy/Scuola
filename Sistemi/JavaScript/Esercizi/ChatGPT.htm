import java.io.*;
import java.util.*;

public class Gestore {
    private List<Figura> figure = new ArrayList<>();

    // Importazione delle figure da un file CSV
    public void importazione(String filename) throws IOException {
        BufferedReader reader = new BufferedReader(new FileReader(filename));
        String line;
        while ((line = reader.readLine()) != null) {
            String[] parts = line.split(",", 2);
            if (parts[0].equals("Quadrilatero")) {
                Quadrilatero quadrilatero = new Quadrilatero(0, 0, 0, 0);
                quadrilatero.fromCsv(line);
                figure.add(quadrilatero);
            } else if (parts[0].equals("Triangolo")) {
                Triangolo triangolo = new Triangolo(0, 0, 0);
                triangolo.fromCsv(line);
                figure.add(triangolo);
            }
        }
        reader.close();
    }

    // Esportazione delle figure in un file CSV
    public void esportazione(String filename) throws IOException {
        BufferedWriter writer = new BufferedWriter(new FileWriter(filename));
        for (Figura figura : figure) {
            writer.write(figura.toCsv());
            writer.newLine();
        }
        writer.close();
    }

    // Stampa il perimetro di tutte le figure
    public void stampaPerimetri() {
        for (Figura figura : figure) {
            System.out.println("Perimetro della figura: " + figura.calcolaPerimetro());
        }
    }
}

public class Triangolo extends Figura {
    private float l1, l2, l3;

    // Costruttore
    public Triangolo(float l1, float l2, float l3) {
        if (!doExist(l1, l2, l3)) {
            throw new IllegalArgumentException("I lati non possono formare un triangolo valido.");
        }
        this.l1 = l1;
        this.l2 = l2;
        this.l3 = l3;
    }

    // Metodo per verificare se i lati possono formare un triangolo valido
    private boolean doExist(float l1, float l2, float l3) {
        return (l1 + l2 > l3) && (l1 + l3 > l2) && (l2 + l3 > l1);
    }

    @Override
    public float area() {
        // Formula di Erone per calcolare l'area di un triangolo
        float s = perimetro() / 2;  // Semi-perimetro
        return (float) Math.sqrt(s * (s - l1) * (s - l2) * (s - l3));
    }

    @Override
    public float perimetro() {
        return l1 + l2 + l3;  // Perimetro del triangolo
    }

    @Override
    public String toString() {
        return "Triangolo: lato1=" + l1 + ", lato2=" + l2 + ", lato3=" + l3 + ", " + descrizione();
    }
}

Classe Triangolo (classe derivata)

La classe Triangolo implementa anche i metodi area() e perimetro(), ma qui utilizziamo la formula di Herone per calcolare l'area, dato che possiamo avere un triangolo qualsiasi.

public class Triangolo extends Figura {
    private float l1, l2, l3;

    // Costruttore
    public Triangolo(float l1, float l2, float l3) {
        if (!doExist(l1, l2, l3)) {
            throw new IllegalArgumentException("I lati non possono formare un triangolo valido.");
        }
        this.l1 = l1;
        this.l2 = l2;
        this.l3 = l3;
    }

    // Metodo per verificare che i lati possano formare un triangolo
    private boolean doExist(float l1, float l2, float l3) {
        return (l1 + l2 > l3) && (l1 + l3 > l2) && (l2 + l3 > l1);
    }

    // Implementazione del metodo area
    @Override
    public float area() {
        float s = perimetro() / 2;  // Semi-perimetro
        return (float) Math.sqrt(s * (s - l1) * (s - l2) * (s - l3));  // Formula di Herone
    }

    // Implementazione del metodo perimetro
    @Override
    public float perimetro() {
        return l1 + l2 + l3;  // Perimetro del triangolo
    }

    @Override
    public String toString() {
        return "Triangolo: lato1=" + l1 + ", lato2=" + l2 + ", lato3=" + l3 + ", " + descrizione();
    }

java
import java.util.LinkedList;
import java.io.BufferedReader;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

public class Gestore {
    private LinkedList<Figura> figure;

    public Gestore() {
        figure = new LinkedList<>();
    }

    public void add(Figura figura) {
        figure.add(figura);
    }

    public String toString() {
        StringBuilder sb = new StringBuilder();
        for (Figura fig : figure) {
            sb.append(fig.toString()).append("\n");
        }
        return sb.toString();
    }

    public String elencoAree() {
        StringBuilder sb = new StringBuilder();
        for (Figura fig : figure) {
            sb.append(fig.getClass().getSimpleName())
              .append(": area=")
              .append(fig.area())
              .append("\n");
        }
        return sb.toString();
    }

    // Esporta tutte le figure in un file CSV
    public void esportaACsv(String filePath) throws IOException {
        FileWriter writer = new FileWriter(filePath);
        for (Figura fig : figure) {
            writer.write(fig.toCsv() + "\n");
        }
        writer.close();
    }

    // Importa figure da un file CSV
    public void importaDaCsv(String filePath) throws IOException {
        BufferedReader reader = new BufferedReader(new FileReader(filePath));
        String line;
        while ((line = reader.readLine()) != null) {
            Figura figura = creaFiguraDaCsv(line);
            if (figura != null) {
                figure.add(figura);
            }
        }
        reader.close();
    }

    // Crea una figura a partire da una stringa CSV
    private Figura creaFiguraDaCsv(String csv) {
        String[] parts = csv.split(",");
        if (parts.length < 2) return null;

        if (parts[0].equals("quadrilatero")) {
            float l1 = Float.parseFloat(parts[1]);
            float l2 = Float.parseFloat(parts[2]);
            return new Quadrilatero(l1, l2);
        } else if (parts[0].equals("triangolo")) {
            float l1 = Float.parseFloat(parts[1]);
            float l2 = Float.parseFloat(parts[2]);
            float l3 = Float.parseFloat(parts[3]);
            return new Triangolo(l1, l2, l3);
        }
        return null;
    }

        if (!isTriangoloValido(lato1, lato2, lato3)) {
            throw new IllegalArgumentException("I lati non possono formare un triangolo valido.");
        }
        this.lato1 = lato1;
        this.lato2 = lato2;
        this.lato3 = lato3;
    }

    private boolean isTriangoloValido(float l1, float l2, float l3) {
        // Verifica se i lati possono formare un triangolo valido
        return (l1 + l2 > l3) && (l1 + l3 > l2) && (l2 + l3 > l1);
    }

    @Override
    public float area() {
        // Formula di Herone per l'area
        float semiperimetro = perimetro() / 2;
        return (float) Math.sqrt(semiperimetro * (semiperimetro - lato1) * (semiperimetro - lato2) * (semiperimetro - lato3));
    }

    @Override
    public float perimetro() {
        return lato1 + lato2 + lato3;
    }

    @Override
    public String toCsv() {
        return "triangolo," + lato1 + "," + lato2 + "," + lato3;
    }

    @Override
    public void fromCsv(String csv) {
        String[] parts = csv.split(",");
        if (parts.length == 4 && parts[0].equals("triangolo")) {
            this.lato1 = Float.parseFloat(parts[1]);
            this.lato2 = Float.parseFloat(parts[2]);
            this.lato3 = Float.parseFloat(parts[3]);
        }
    }

    @Override
    public String toString() {
        return "Triangolo: lato1=" + lato1 + ", lato2=" + lato2 + ", lato3=" + lato3 + ", area=" + area() + ", perimetro=" + perimetro();
    }
}

5. Classe Gestore

La classe Gestore si occupa di gestire l'elenco delle figure e offre metodi per importare ed esportare da e verso file CSV. Essa permette anche di stampare il perimetro di ogni figura, utilizzando il polimorfismo.

java
import java.io.*;
import java.util.*;

public class Gestore {
    private List<Figura> figure;

    public Gestore() {
        figure = new ArrayList<>();
    }

    // Aggiunge una figura alla lista
    public void add(Figura figura) {
        figure.add(figura);
    }

    // Stampa il perimetro di tutte le figure
    public void stampaPerimetri() {
        for (Figura fig : figure) {
            System.out.println(fig.getClass().getSimpleName() + " - Perimetro: " + fig.getPerimetro());
        }
    }

    // Esporta tutte le figure in un file CSV
    public void esportaACsv(String filePath) throws IOException {
        BufferedWriter writer = new BufferedWriter(new FileWriter(filePath));
        for (Figura figura : figure) {
            writer.write(figura.toCsv());
            writer.newLine();
        }
        writer.close();
    }

    // Importa le figure da un file CSV
    public void importaDaCsv(String filePath) throws IOException {
        BufferedReader reader = new BufferedReader(new FileReader(filePath));
        String line;
        while ((line = reader.readLine()) != null) {
            Figura figura = creaFiguraDaCsv(line);
            if (figura != null) {
                figure.add(figura);
            }
        }
        reader.close();
    }

    // Crea una figura a partire da una stringa CSV
    private Figura creaFiguraDaCsv(String csv) {
        String[] parts = csv.split(",");
        if (parts.length < 2) return null;

        if (parts[0].equals("quadrilatero")) {
            float l1 = Float.parseFloat(parts[1]);
            float l2 = Float.parseFloat(parts[2]);
            return new Quadrilatero(l1, l2);
        } else if (parts[0].equals("triangolo")) {
            float l1 = Float.parseFloat(parts[1]);
            float l2 = Float.parseFloat(parts[2]);
            float l3 = Float.parseFloat(parts[3]);
            return new Triangolo(l1, l2, l3);
        }
        return null;
    }
}
